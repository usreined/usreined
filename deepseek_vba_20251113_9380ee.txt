// 带简单响应的聊天脚本

Function GetResponseMessage(keyword)
    Dim response
    Select Case LCase(keyword)
        Case "你好", "hello", "hi"
            response = "你好！很高兴见到你"
        Case "时间", "time"
            response = "现在时间是：" & Now()
        Case "天气", "weather"
            response = "今天天气看起来不错"
        Case Else
            response = "我看到了你的消息"
    End Select
    GetResponseMessage = response
End Function
